\begin{algorithm}[htbp]
  \begin{algorithmic}
    \Procedure{bubblesort}{\,\(a_{1}, \dotsc, a_{n}\) : reelle tal med \(n \geq 2\)\,}
    \For{\(i \coloneqq 1\) til \(n - 1\)}
    \For{\(j \coloneqq 1\) til \(n - i\)}
    \If{\(a_{j} > a_{j+1}\)}
    ombyt \(a_{j}\) og \(a_{j+1}\)
    \EndIf
    \EndFor
    \EndFor
    \EndProcedure
    \Comment{\(a_{1}, \dotsc, a_{n}\) er nu i voksende orden}
  \end{algorithmic}
  \caption{Bubble Sort}
  \label{alg:some}
\end{algorithm}
